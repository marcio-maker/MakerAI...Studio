import { MeshEngine } from '@makerai/mesh-engine';

const mesh = new MeshEngine({
    container: '#meshCanvas',
    resolution: 1920,
    fps: 60,
    gradient: {
        colors: ['#A855F7', '#F97316'],
        intensity: 0.5,
        animation: 'flow'
    }
});

// Start rendering
mesh.start();

// Update gradient in real-time
mesh.updateGradient({
    colors: ['#3B82F6', '#10B981'],
    intensity: 0.8
});
