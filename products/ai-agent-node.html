Deployment_Script.sh 
#!/bin/bash

# AI Agent Node Installation
echo "Initializing AI Node..."

# 1. Pull Docker container
docker pull makerai/agent-node:v3.2.1

# 2. Generate keys
node -e "require('crypto').generateKeyPairSync('rsa', {
  modulusLength: 4096,
  publicKeyEncoding: { type: 'spki', format: 'pem' },
  privateKeyEncoding: { type: 'pkcs8', format: 'pem' }
})"

# 3. Run node
docker run -d \
  --name ai-agent-node \
  -p 3000:3000 \
  -v /data/ai-node:/app/data \
  makerai/agent-node:v3.2.1

# 4. Connect to network
curl -X POST http://localhost:3000/network/join \
  -H "Authorization: Bearer ${NODE_TOKEN}"

echo "✅ AI Agent Node deployed successfully!"
